    p2.*, 
    p1.par001_desc 
FROM "{{JSutils.currentSchema()}}".hotamigo001 p2
INNER JOIN "{{JSutils.currentSchema()}}".hotamigo001 p1 
    ON p1.amigo001_id = p2.amigo001_id
