SELECT 
    amigo001.*,
    p1.ani002_id,
    p1.ani002_nome
FROM "{{JSutils.currentSchema()}}".hotamigo001 amigo001
INNER JOIN "{{JSutils.currentSchema()}}".hotani002 p1 
    ON p1.ani002_id = amigo001.amigo001_ani002_id_amigo
WHERE amigo001.amigo001_deleted <> true 
AND amigo001.amigo001_ani002_id_animal = {{ appsmith.store.petID_Amigos }}
ORDER BY p1.ani002_nome ASC;
